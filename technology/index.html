Fabric Workspace Provisioner (Python)

import azure.identity
from azure.mgmt.fabric import FabricManagementClient

def provision_fabric_workspace(subscription_id, resource_group, workspace_name):
    credential = DefaultAzureCredential()
    client = FabricManagementClient(credential, subscription_id)
    
    workspace = client.workspaces.create_or_update(
        resource_group_name=resource_group,
        workspace_name=workspace_name,
        workspace={
            "location": "UK South",
            "properties": {
                "description": "Self-service Fabric workspace"
            }
        }
    )
    return workspace

Iceberg Cross-Cloud Sync (Python)

from pyiceberg.catalog import load_catalog
from pyiceberg.table import Table

def sync_iceberg_table(source_catalog, target_catalog, table_name):
    source_table = load_catalog(source_catalog).load_table(table_name)
    target_table = load_catalog(target_catalog).load_table(table_name)
    
    # Sync metadata and data
    target_table.overwrite(source_table.to_pandas())
    print(f"Synced {table_name} from {source_catalog} to {target_catalog}")
